taxsim.js demo

requirements:
  • taxsim.js
  • taxsim.wasm

usage

  1. place both files on a webserver
  2. include taxsim.js from an html page
  3. invoke the taxsim() javascript function
  4. resolve the returned Promise to retrieve the stdout/stderr outputs
<script src="taxsim.js"></script>
<script>
  // taxsim.js implements a function with the following prototype:
  //
  //    function taxsim(input: String): Promise
  //
  // the returned promise can be manipulated either directly or via async/await

  // sample input
  let input = ["taxsimid,mstat,year,ltcg,idtl", "1,2,1970,100000,5"].join("\n")

  // promise api
  taxsim(input).then(function(output) {
    console.log('taxsim stdout', output)
  }).catch(function(error) {
    console.log('taxsim stderr', error)
  })

  // async/await api
  try {
    let output = await taxsim(input)
    console.log('taxsim stdout', output)
  } catch(error) {
    console.log('taxsim stderr', error)
  }
</script>
